using namespace std;

char G[2][2000]; 
int n;
queue<pi> q;

int f(int & ret){
	ret = 0; bool c = 0;
	loop(i,2) loop(j,n)
	if(G[i][j] == '.'){
		int k = j;
		while(k < n && G[i][k] != 'X') k++;
		if(k - j == 1) q.push(mp(i,j)),c = 1;
		ret ++;
		j = k - 1;
	}
	return c;
}

int g(){
	int ret = 0;
	while(!q.empty()){
		int i = q.front().xx,j = q.front().yy; q.pop();
		if(G[i][j] != '.') continue;
		ret++;
		G[i][j] = 'o';
		if(G[i][j] == '#') continue;
		for(int k = j + 1;k < n;k++) {
			if(G[!i][k] == 'X') break;		
			G[!i][k] = 'o';
		}
		for(int k = j; k >= 0;k--) {
			if(G[!i][k] == 'X') break;		
			G[!i][k] = 'o';
		}
	}
	return ret;
}

int main(){
	int T;
	scanf("%d",&T);
	loop(t,T){
		scanf("%d %s %s",&n,G[0],G[1]);
		int ans = 0,x;
		f(x); 
		ans = g();
		f(x);
		ans += x;
		printf("Case #%d: %d\n",t + 1,ans);
	}	
	return 0;
}
